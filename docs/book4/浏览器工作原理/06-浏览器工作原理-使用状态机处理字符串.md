# 浏览器工作原理-使用状态机处理字符串

## 使用有限状态机处理字符串

- 在一个字符串中，找到字符'abcdefg'

```
  function match(string) {
    let state = start
    for(let key of string) {
      state = state(key)
    }
    return state === end
  }

  function start(key) {
    if (key === 'a') {
      return foundA
    } else {
      return start
    }
  }

  function end(key) {
    return end
  }

  function foundA(key) {
    if (key === 'b') {
      return foundB
    } else {
      return start(key)
    }
  }

  function foundB(key) {
    if (key === 'c') {
      return foundC
    } else {
      return start(key)
    }
  }

  function foundC(key) {
    if (key === 'd') {
      return foundD
    } else {
      return start(key)
    }
  }

  function foundD(key) {
    if (key === 'e') {
      return foundE
    } else {
      return start(key)
    }
  }

  function foundE(key) {
    if (key === 'f') {
      return foundF
    } else {
      return start(key)
    }
  }

  function foundF(key) {
    if (key === 'g') {
      return end
    } else {
      return start(key)
    }
  }

  match('abcdefghijk')
  match('aabsdaesaa')
  // 状态在经过ab后，又进入了start中，而值没有重置
  // 通过返回start(key)来进行值的重置
  match('ababcdefg')
```
